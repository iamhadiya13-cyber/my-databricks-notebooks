  INTO ODI_TRG.TRG_EMPLOYEES TRG_EMPLOYEES
  USING
  (
SELECT 
  SRC_EMPLOYEES.EMPLOYEE_ID AS EMPLOYEE_ID ,
  SRC_EMPLOYEES.FIRST_NAME AS FIRST_NAME ,
  SRC_EMPLOYEES.LAST_NAME AS LAST_NAME ,
  SRC_EMPLOYEES.EMAIL AS EMAIL ,
  SRC_EMPLOYEES.PHONE_NUMBER AS PHONE_NUMBER ,
  SRC_EMPLOYEES.HIRE_DATE AS HIRE_DATE ,
  SRC_EMPLOYEES.JOB_ID AS JOB_ID ,
  SRC_EMPLOYEES.SALARY AS SALARY ,
  SRC_EMPLOYEES.COMMISSION_PCT AS COMMISSION_PCT ,
  SRC_EMPLOYEES.MANAGER_ID AS MANAGER_ID ,
  SRC_EMPLOYEES.DEPARTMENT_ID AS DEPARTMENT_ID ,
  SRC_EMPLOYEES.LAST_UPD_DT AS LAST_UPD_DT   
FROM 
  ODI_SRC.SRC_EMPLOYEES@LS_ODI_SRC SRC_EMPLOYEES  
WHERE
  (SRC_EMPLOYEES.LAST_UPD_DT >= SYSDATE
) 
  ) MERGE_SUBQUERY
  ON
  (
    TRG_EMPLOYEES.EMPLOYEE_ID = MERGE_SUBQUERY.EMPLOYEE_ID 
   )
  WHEN NOT MATCHED THEN
    INSERT
    (
      EMPLOYEE_ID ,
      FIRST_NAME ,
      LAST_NAME ,
      EMAIL ,
      PHONE_NUMBER ,
      HIRE_DATE ,
      JOB_ID ,
      SALARY ,
      COMMISSION_PCT ,
      MANAGER_ID ,
      DEPARTMENT_ID ,
      LAST_UPD_DT

      MERGE_SUBQUERY.FIRST_NAME ,

      MERGE_SUBQUERY.EMAIL ,
      MERGE_SUBQUERY.PHONE_NUMBER ,

      MERGE_SUBQUERY.MANAGER_ID ,
      MERGE_SUBQUERY.DEPARTMENT_ID ,
      MERGE_SUBQUERY.LAST_UPD_DT 
    )
